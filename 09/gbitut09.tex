	\begin{frame}{Rückblick: Kontextfreie Grammatiken}
		\begin{itemize}[<+->]
			\item Ein Vier-Tupel: $G = (N, T, S, P)$
			\item Produktionen definieren Ersetzungen eines Nichtterminals mit Wörtern über $N \cup T$
			\item Wir wenden Produktionen in Ableitungsschritten an: $v \Rightarrow w, v \Rightarrow^* w$
			\item $L(G)$ sind alle aus $S$ ableitbaren Wörter über $T$ (die also nur aus Terminalsymbolen bestehen)
		\end{itemize}
	\end{frame}
}

\begin{frame}{Wahr oder Falsch?}
	\thasse{\Socrative}
	Sei $G=(\{X,Y\},\{a,b\},X,P)$ eine kontextfreie Grammatik.
	\FalseQuestion{Die Produktion $XY \to a$ ist gültig}
	\FalseQuestion{Die Produktion $a \to XY$ ist gültig}
	\TrueQuestionE{Die Produktion $X \to XaX$ ist gültig}{}
	\TrueQuestionE{Wenn $X \to w$ eine gültige Produktion ist, dann gilt $X \Rightarrow w$}{}
	\FalseQuestionE{Wenn $X \Rightarrow w$ gilt, dann ist $X \to w$ eine Produktion in P }{Sei $P=\{X \to XX \mid a\}$. Dann gilt $XX \Rightarrow Xa$, aber $XX \to Xa \notin P$}	
\end{frame}
