	\begin{frame}[t]{Wahr oder Falsch?}
		\thasse{\Socrative}
		\FalseQuestionE{Die Korrektheit eines Algorithmus kann man durch Testen \\ beweisen.}{Testen kann nur Fehler aufzeigen, aber nicht Fehlerfreiheit garantieren.}
		\FalseQuestionE{Sinnvolle Schleifeninvarianten kann man \enquote{nach Kochrezept} aufstellen.}{Das Aufstellen sinnvoller Schleifeninvarianten erfordert viel Übung und kann insbesondere \textbf{nicht} vom Rechner gelöst werden.}
		\FalseQuestionE{Ein Algorithmus ist ein (kompilierbares/ausführbares) Programm.}{Der Algorithmus selbst ist nur die Beschreibung der (Rechen-)\\ Vorschriften, nicht die tatsächliche Umsetzung in Programmcode.}
		\TrueQuestion{$x = y \aland y = z \impl x = z$ ist allgemeingültig}
		\FalseQuestion{$x = z \impl x = y \aland y = z$ ist allgemeingültig}
	\end{frame}
}

		\item \textbf{A4.5}: MuLö war falsch (Übung auch!), wurde aktualisiert. \\
			  \textbf{Aufgabe}: \; AL-Fml. $G$ in PL-Fml. $P$ bedeutungserhaltend übersetzen. \\ 
			  \pause
			  \textbf{Falsche MuLö}: \; $I(\word c) = \W$ \\
			  \quad Setze $P := G \pleq \word c$ \qquad Bsp.: \; $G = \plka \word A \aland \alnot \word B \plkz \impl P := \plka \word A \aland \alnot \word B \plkz \pleq \word c$ \\ 
				  \pause 
				  \quad \impl Kaputt, weil nur Terme in $\pleq$ eingesetzt werden dürfen! \\
			  \pause
			  \textbf{Möglicher Fix}: \; Ersetze jede AL-Variable durch Vergleich mit \word c: \\
			  \quad $G = \word A \aland \alnot \word B \; \?> \; P := \plka \word A \pleq \word c\plkz \aland \alnot \plka \word B \pleq \word c\plkz$ \\
			  Alternativ: S. aktualisierte MuLö.
	\end{itemize}
\end{frame}
